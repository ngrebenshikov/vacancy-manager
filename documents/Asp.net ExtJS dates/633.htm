ExtJS: Валидация двух datefield для интервала времени

22.03.2012 в 16:05 admin Комментариев нет

Задача: есть два datefield на форме и необходимо, чтобы в первом, где выбирается начальная дата, нельзя было выбрать значение, превышающее дату окончания. И наоборот.
Решение:

показать исходник
печатать?
01    Ext.apply(Ext.form.field.VTypes, {
02        daterange: function (val, field) {
03            var date = field.parseDate(val);
04            if (!date) { return false; }
05            if (field.startDateField && (!this.dateRangeMax || (date.getTime() != this.dateRangeMax.getTime()))) {
06                var start = field.up('form').down('#' + field.startDateField);
07                start.setMaxValue(date);
08                start.validate();
09                this.dateRangeMax = date;
10            }
11            else if (field.endDateField && (!this.dateRangeMin || (date.getTime() != this.dateRangeMin.getTime()))) {
12                var end = field.up('form').down('#' + field.endDateField);
13                end.setMinValue(date);
14                end.validate();
15                this.dateRangeMin = date;
16            }
17            return true;
18        },
19        daterangeText: 'Начальная дата должна быть меньше конечной'
20    });
21 
22// способ применения (в конфиге формы, например)

24    xtype: 'datefield',
25    startDay: 1,
26    fieldLabel: 'Начало периода',

28    vtype: 'daterange',
29    endDateField: 'enddt'
